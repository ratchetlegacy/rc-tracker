-- Supprimer l'ancienne table si elle existe
DROP TABLE IF EXISTS items;

CREATE TABLE items (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE NOT NULL,
  name TEXT NOT NULL,
  type TEXT DEFAULT '',
  edition TEXT DEFAULT '',
  manufacturer TEXT DEFAULT '',
  platform TEXT DEFAULT '',
  region TEXT DEFAULT 'PAL',
  condition TEXT DEFAULT 'occasion',
  status TEXT NOT NULL DEFAULT 'owned',
  has_box BOOLEAN DEFAULT false,
  has_manual BOOLEAN DEFAULT false,
  has_disc BOOLEAN DEFAULT false,
  has_sleeve BOOLEAN DEFAULT false,
  has_docs BOOLEAN DEFAULT false,
  buy_price NUMERIC(10,2),
  cur_value NUMERIC(10,2),
  img_url TEXT DEFAULT '',
  rating INTEGER DEFAULT NULL,
  notes TEXT DEFAULT '',
  wish_url TEXT DEFAULT '',
  comment TEXT DEFAULT '',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

ALTER TABLE items ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users see own items" ON items FOR ALL
  USING (auth.uid() = user_id)
  WITH CHECK (auth.uid() = user_id);

📖 HOW TO USE

RATCHET & CLANK · COLLECTION TRACKER

How to use the Tracker

Everything you need to manage, track and show off your collection.

📱 Install the app on your phone
🍎

iPhone / iPad (Safari)

Open the tracker in Safari. Tap the Share button (the square with an arrow, at the bottom), scroll down and tap Add to Home Screen, then Add. The R&C Tracker icon appears on your home screen and opens full-screen like a real app.

🤖

Android (Chrome)

Open the tracker in Chrome. Tap the ⋮ menu (top right), then Add to Home screen (or Install app), and confirm. The icon is added and the app launches full-screen, offline-ready.

👤 For collectors
➕

Add an item

Tap the + ADD OBJECT button (bottom right). Fill in the name, type, platform, edition and region. Set its status — Owned or Wishlist — its condition (New/Sealed or Second hand), and optionally a buy price and current market value. Pick a cover from the image bank, then save.

🖼️

Choose a cover

In the add/edit form, tap Image bank to browse covers sorted by folder (PS2, PS3, Figures…). Use the search box to filter by name. Tap an image then Choose. Can't find yours? Use Submit a photo to send one to the admin for review.

✅

Completeness

For games, tick what you have (box, manual, disc…). The app tracks which items are complete and flags those missing pieces, so you always know what's left to find.

🎯

Wishlist

Set an item's status to Wishlist for things you're hunting. They show up separately and don't count as owned, but appear in your stats and can carry a target price.

📊

Stats

Open Stats from the menu to see your KPIs (owned, wishlist, invested, market value, profit/loss), your global progress toward the full R&C collection, distribution by type and region, condition, your most valuable items, and item completeness.

🏆

Ranks

The more you own, the higher your rank climbs — from Galactic Ranger Recruit all the way to Insomniac Legend. Your rank and progress to the next one show on your profile and the home header.

📄

Export to PDF

From the menu, choose Export PDF. Pick a scope (Owned / Wishlist / Everything) then a format: Simple Table (light list) or Magazine PDF (cover, profile, full stats and your items by category with covers).

🤝

Trades

The Trades section lets collectors propose items for exchange. Browse what others offer and put your own duplicates up for trade.

🛠️ For the admin
☁️

Import from Cloudinary

In Admin → Image bank, hit Import from Cloudinary. It pulls every cover from your Cloudinary Jaquettes folders, sorts them into the right folder automatically, and fixes folders on images already imported.

✏️

Rename / fix an image

Each image in the bank has a ✏️ button to change its name, folder or region — handy when an import didn't classify something the way you want.

📥

Photo submissions

When a collector submits a photo, it lands in your review queue. Approve to add it to the shared bank, or Reject it.

🌍

Collection total

Set the total number of known R&C items (e.g. 2500) in the admin panel. This drives the global progress bar and the Collection stat — and it's shared for every user, updating instantly.
