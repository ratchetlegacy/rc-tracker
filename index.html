📋 Script SQL à coller dans Supabase → SQL Editor 
-- Table principale des items
CREATE TABLE items (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE NOT NULL,
  name TEXT NOT NULL,
  cat TEXT NOT NULL DEFAULT 'jeu',
  status TEXT NOT NULL DEFAULT 'owned',
  condition TEXT DEFAULT 'new',
  lang TEXT DEFAULT 'FR',
  has_box BOOLEAN DEFAULT false,
  has_cd BOOLEAN DEFAULT false,
  has_manual BOOLEAN DEFAULT false,
  buy_price NUMERIC(10,2),
  cur_value NUMERIC(10,2),
  note TEXT DEFAULT '',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Sécurité Row Level Security (chaque user voit ses données uniquement)
ALTER TABLE items ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users see own items"
  ON items FOR ALL
  USING (auth.uid() = user_id)
  WITH CHECK (auth.uid() = user_id);
